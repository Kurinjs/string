using System;

using System.Linq;

namespace rekyrsia
{
    class Program
    {
        static void Main(string[] args)
        {
            int num = 5,num1=num;
            num = rek(num);

            Console.WriteLine("Факториал числа {0} = {1}", num1, num);


            Console.ReadLine();
        }
        public static int rek(int n)
        {
            if (n == 1)
            {
                return 1;
            }
            return rek(n - 1) * n;
        }
